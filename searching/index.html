# binary search
def binarySearch(alist, item):
    first = 0
    last = len(alist)-1
    found = False
    while first<=last and not found:
        midpoint = (first + last)//2
        if alist[midpoint] == item:
            found = True
        else:
            if item < alist[midpoint]:
                last = midpoint-1
            else:
                first = midpoint+1
    return found

Perbandingan Linear vs Binary Search¶

import random

# =========================
# Linear Search
# =========================
def linear_search(data, target):
    iterasi = 0
    for i in range(len(data)):
        iterasi += 1
        if data[i] == target:
            return iterasi
    return iterasi  # tetap return jumlah iterasi walau tidak ditemukan


# =========================
# Binary Search
# =========================
def binary_search(data, target):
    left = 0
    right = len(data) - 1
    iterasi = 0

    while left <= right:
        iterasi += 1
        mid = (left + right) // 2

        if data[mid] == target:
            return iterasi
        elif data[mid] < target:
            left = mid + 1
        else:
            right = mid - 1

    return iterasi


# =========================
# Data Random (100 elemen)
# =========================

# Target sesuai kondisi

def uji(nama, target):
    print(f"\n=== {nama} (target = {target}) ===")

    it_lin_acak = linear_search(data, target)
    it_lin_urut = linear_search(data_sorted, target)
    it_bin = binary_search(data_sorted, target)

    print("Linear (tidak terurut):", it_lin_acak, "iterasi")
    print("Linear (terurut):     ", it_lin_urut, "iterasi")
    print("Binary Search:        ", it_bin, "iterasi")
