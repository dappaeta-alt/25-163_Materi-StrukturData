    count=ind
    found=False
    while (count!=ind-1) and not(found):
        if hashTable[count]=='none':
            found=True
            hashTable[count]=data
        else:
            count=count+1
            if count==len(hashTable)-1:

def midSqFunction(data,table_size):
    # Mengkuadratkan data dan mengubahnya menjadi string
    square = str(data ** 2)

    # menentukan titik tengah
    midInd = len(square) // 2

    # mengambil bagian tengah string
    if len(square) < 3:
        # jika hasil kuadrat kurang dari 3 digit, ambil seluruhnya
        midVal = int(square)
    elif len(square) % 2 == 0:
        # Jika jumlah digit genp, ambil 2 digit tengah
        midVal = int(square[midInd-1:midInd+1])
    else:
        # jika jumlah digit ganjil, ambil 3 digit tepat tengah
        midVal = int(square[midInd-1:midInd+2])

    return midVal % table_size

    ind = hashFunction(data,len(table))
    table[ind].append(data)
    print(f"Insert: {data} dimasukkan ke slot {ind}")
    return table

# read
def searchChaining(data, table):
    ind = hashFunction(data, len(table))

    n = len(table[ind])
    for i in range(n):
        if data == table[ind][i]:
            print(f"Search: {data} ditemukan di slot {ind}")
            return True
        print(f"Search: {data} tidak ditemukan")
        return False

# Delete (Search dan Delete)
def deleteChaining(data, table):
    ind = hashFunction(data, len(table))

    n = len(table[ind])
    for i in range(n):
        if data == table[ind][i]:
            table[ind].pop(i)
            print(f"Delete: {data} berhasil dihapus dari slot {ind}")
            return True
        print(f"Delete: {data} gagal dihapus (data tidak ditemukan)")
        return False

def insertLinear(data, table):

    return -1

class HashChaining:
    def __init__(self, size):
        self.size = size
        self.table = [[] for _ in range(size)]

    def hash_function(self, key):
        return key % self.size

    # CREATE
    def insert(self, key):
        index = self.hash_function(key)
        if key not in self.table[index]:
            self.table[index].append(key)

    # READ
    def search(self, key):
        index = self.hash_function(key)
        return key in self.table[index]

    # UPDATE
    def update(self, old, new):
        index = self.hash_function(old)
        if old in self.table[index]:
            self.table[index].remove(old)
            self.insert(new)

    # DELETE
    def delete(self, key):
        index = self.hash_function(key)
        if key in self.table[index]:
            self.table[index].remove(key)

    def display(self):
        print(self.table)
class LinearProbing:
    def __init__(self, size):
        self.size = size
        self.table = [None] * size

    def hash_function(self, key):
        return key % self.size

    def insert(self, key):
        index = self.hash_function(key)
        start = index

        while self.table[index] is not None:
            index = (index + 1) % self.size
            if index == start:
                print("Tabel penuh")
                return
        self.table[index] = key

    def search(self, key):
        index = self.hash_function(key)
        start = index

        while self.table[index] is not None:
            if self.table[index] == key:
                return True
            index = (index + 1) % self.size
            if index == start:
                return False
        return False

    def update(self, old, new):
        if self.delete(old):
            self.insert(new)

    def delete(self, key):
        index = self.hash_function(key)
        start = index

        while self.table[index] is not None:
            if self.table[index] == key:
                self.table[index] = None
                return True
            index = (index + 1) % self.size
            if index == start:
                return False
        return False

    def display(self):
        print(self.table)

class MidSquareHash:
    def __init__(self, size):
        self.size = size
        self.table = [None] * size

    def midsquare_hash(self, key):
        square = key * key
        mid = len(str(square)) // 2
        mid_digits = int(str(square)[mid-1:mid+1])
        return mid_digits % self.size

    def insert(self, key):
        index = self.midsquare_hash(key)
        while self.table[index] is not None:
            index = (index + 1) % self.size
        self.table[index] = key

    def search(self, key):
        index = self.midsquare_hash(key)
        start = index

        while self.table[index] is not None:
            if self.table[index] == key:
                return True
            index = (index + 1) % self.size

                return False
