def insertion_sort(arr):
    n = len(arr)
    
    for i in range(1, n):
        key = arr[i]
        j = i - 1
        shifting = 0
        
        print(f"\nIterasi ke-{i}")
        print(f"Key = {key}")
        
        # Proses shifting
        while j >= 0 and arr[j] > key:
            arr[j + 1] = arr[j]
            j -= 1
            shifting += 1
        
        # Menempatkan key
        arr[j + 1] = key
        
        print(f"Posisi akhir key = index {j+1}")
        print(f"Jumlah shifting = {shifting}")
        print(f"Array sekarang = {arr}")

# BUBBLE SORT DUA ARAH (MODIFIKASI)
def bubble_dua_arah(arr):
    data = arr[:]
    left = 0
    right = len(data) - 1
    iterasi = 1
    total_swap = 0

    while left < right:
        swapped = False

        if iterasi % 2 == 1:
            print("\nITERASI KE-", iterasi, "(GANJIL kiri ke kanan)")
            for i in range(left, right):
                print("(", data[i], "vs", data[i+1], ")", end=" ")

                if data[i] > data[i+1]:
                    data[i], data[i+1] = data[i+1], data[i]
                    total_swap += 1
                    swapped = True
                    print("swap", data)
                else:
                    print("tidak swap")

            right -= 1

        else:
            print("\nITERASI KE-", iterasi, "(GENAP kanan ke kiri)")
            for i in range(right, left, -1):
                print("(", data[i], "vs", data[i-1], ")", end=" ")

                if data[i] < data[i-1]:
                    data[i], data[i-1] = data[i-1], data[i]
                    total_swap += 1
                    swapped = True
                    print("swap", data)
                else:
                    print("tidak swap")

            left += 1

        print("Hasil Iterasi", iterasi, ":", data)

        if not swapped:
            print("Early stopping aktif berhenti")
            break

        iterasi += 1

    print("Total iterasi:", iterasi)
    print("Total swap:", total_swap)
    return data

# BUBBLE SORT BIASA
def bubble_biasa(arr):
    data = arr[:]
    n = len(data)
    iterasi = 0
    total_swap = 0

    for i in range(n):
        iterasi += 1
        swapped = False

        for j in range(0, n - i - 1):
            if data[j] > data[j + 1]:
                data[j], data[j + 1] = data[j + 1], data[j]
                total_swap += 1
                swapped = True

        if not swapped:
            break

    print("\nBubble Biasa")
    print("Total iterasi:", iterasi)
    print("Total swap:", total_swap)
    return data

# SELECTION SORT MIN MAX
def selection_min_max(arr):
    data = arr[:]
    left = 0
    right = len(data) - 1
    iterasi = 1
    total_swap = 0

    while left <= right:
        print("\nITERASI KE-", iterasi)

        if iterasi % 2 == 1:
            min_idx = left
            for i in range(left, right + 1):
                if data[i] < data[min_idx]:
                    min_idx = i

            if min_idx != left:
                data[left], data[min_idx] = data[min_idx], data[left]
                total_swap += 1

            print("Cari MIN:", data)
            left += 1

        else:
            max_idx = left
            for i in range(left, right + 1):
                if data[i] > data[max_idx]:
                    max_idx = i

            if max_idx != right:
                data[right], data[max_idx] = data[max_idx], data[right]
                total_swap += 1

            print("Cari MAX:", data)
            right -= 1

        iterasi += 1

    print("Total iterasi:", iterasi - 1)
    print("Total swap:", total_swap)
    return data

def selection_biasa(arr):

    n = len(data)
    iterasi = 0
    total_swap = 0

    for i in range(n):
        iterasi += 1
        min_idx = i

        for j in range(i + 1, n):
            if data[j] < data[min_idx]:
                min_idx = j

        if min_idx != i:
            data[i], data[min_idx] = data[min_idx], data[i]
            total_swap += 1

    print("\nSelection Biasa")

    print("Total swap:", total_swap)
    return data

# INSERTION SORT MODIFIKASI
def insertion_modifikasi(arr):
    data = arr[:]
    iterasi = 0
    shifting = 0

    for i in range(1, len(data)):
        iterasi += 1
        key = data[i]

        if key >= data[i - 1]:
            continue

        j = i - 1
        while j >= 0 and data[j] > key:
            data[j + 1] = data[j]
            shifting += 1
            j -= 1

        data[j + 1] = key

    print("\nInsertion Modifikasi")
    print("Total iterasi:", iterasi)
    print("Total shifting:", shifting)
    return data
